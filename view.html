function checkProName(app_name_obj,app_id_obj,lang_obj){
var lang;
if(lang_obj){
if(typeof(lang_obj) == 'string'){
lang = lang_obj;
}else{
lang = lang_obj.val();
}
}else{
lang = '';
}
var app_name = app_name_obj.val();
var reStatus;
$.ajax({
type: "POST",
cache: false,
async: false,
dataType:"json",
data: {app_name:app_name,lang:lang},
url: "/Git/testadmin/manage.php?s=/U3dFileManageB/serarch_app_byname.html",
success: function(data) {
if(data.status == 1){
app_id_obj.val(data.app_id);
reStatus=1;
}else{
alert('该语言未查到这款产品，请核对');
app_id_obj.val('');
reStatus=0;
}
}
});
return reStatus;
}
function autoComplete_pre(app_name_obj,app_id_obj,lang_obj){

function autoComplete(app_name_obj,app_id_obj,lang) {

},
useCache: false,
filterResults: false,
extraParams:{lang:lang}
});
}
